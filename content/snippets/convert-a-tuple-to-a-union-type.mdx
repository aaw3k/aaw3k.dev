---
title: Convert a Tuple to a Union Type
description: TypeScript lets you create a union type from a tuple type. That's useful when you want to make a type that's a union of multiple types.
date: '2022-01-30'
tags: ['TypeScript']
---

```ts
type UnionFromTuple<Tuple extends readonly (string | number | boolean)[]> =
  Tuple[number];

const animals = ['🐶', '🙀', '🐮'] as const;

type Animal = UnionFromTuple<typeof animals>;

//  type Animal = '🐶' | '🙀' | '🐮';
```
